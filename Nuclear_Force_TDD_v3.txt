1. System Overview
Nuclear Force is a fully client-side, single-page web app. There is no backend. All blast geometry, fallout calculations, and shelter routing run in the browser using Turf.js. Deployed on Replit. No server-side processing required.


1.1 Core User Flow (3 steps)
1. User clicks 'Place Blast' and taps the map to set the detonation epicenter.
2. User enters their address in the search field, or clicks GPS to auto-locate.
3. User hits 'Analyze' — blast zones render, nearest shelter appears in the right panel, walking directions populate.


1.2 Updated Data Flow [UPDATED]

     ↓

5.2 gps.ts — GPS + Address Geocoding [UPDATED]
Now handles both GPS acquisition and free-text address geocoding via Nominatim (OSM geocoder, no API key required).


// Nominatim geocoding — free, no key needed
export async function geocodeAddress(query: string): Promise<UserPosition> {
  const url = `https://nominatim.openstreetmap.org/search?q=${encodeURIComponent(query)}&format=json&limit=1`;
  const res  = await fetch(url, { headers: { 'Accept-Language': 'en' } });
  const data = await res.json();
  if (!data.length) throw new Error('Address not found');
  return {
    lat: parseFloat(data[0].lat),
    lng: parseFloat(data[0].lon),
    label: data[0].display_name
  };
}


export async function getUserPosition(): Promise<UserPosition> {
  return new Promise((resolve, reject) => {
    navigator.geolocation.getCurrentPosition(
      ({ coords }) => resolve({ lat: coords.latitude, lng: coords.longitude }),
      (err) => reject(err),
      { enableHighAccuracy: true, timeout: 8000, maximumAge: 0 }
    );
  });
}


Nominatim usage policy: max 1 request/second, must include a valid User-Agent header. For a hackathon this is fine. Add 'User-Agent': 'NuclearForce/1.0' to all fetch headers.


5.3 shelters.ts — Nearest Shelter Engine


import * as turf from '@turf/turf';
import shelterData from './data/shelters.geojson';


export function getNearestShelter(userPos: UserPosition) {
  const userPt = turf.point([userPos.lng, userPos.lat]);
  return turf.nearestPoint(userPt, shelterData as any);
}


export function getDistanceMeters(userPos: UserPosition, shelter: any): number {
  const from = turf.point([userPos.lng, userPos.lat]);
  const to   = turf.point(shelter.geometry.coordinates);
  return Math.round(turf.distance(from, to, { units: 'meters' }));
}


export function getWalkingTimeMin(distanceM: number): number {
  return Math.ceil((distanceM / 1000 / 5) * 60); // 5 km/h
}




5.4 threat.ts — Blast Radius + Fallout Engine


export function drawBlastZones(map: L.Map, config: BlastConfig): L.LayerGroup {
  const radii = BLAST_RADII[config.yieldPreset];
  const epi   = turf.point([config.epicenterLng, config.epicenterLat]);
  const group = L.layerGroup();
  const zones = [
    { r: radii.lightBlast,  color: '#F1C40F', label: 'Light Blast'  },
    { r: radii.heavyBlast,  color: '#E67E22', label: 'Heavy Blast'  },
    { r: radii.fireball,    color: '#E74C3C', label: 'Fireball'     },
  ];
  zones.forEach(z => {
    L.geoJSON(turf.circle(epi, z.r, { units:'kilometers' }), {
      style: { color: z.color, fillColor: z.color, fillOpacity: 0.25, weight: 2 }
    }).addTo(group);
  });
  return group.addTo(map);
}


export function drawFalloutPlume(map: L.Map, config: BlastConfig): L.Polygon {
  const epi   = turf.point([config.epicenterLng, config.epicenterLat]);
  const tip   = turf.destination(epi, 18, config.windBearing, { units:'kilometers' });
  const left  = turf.destination(epi, 4,  config.windBearing + 90, { units:'kilometers' });
  const right = turf.destination(epi, 4,  config.windBearing - 90, { units:'kilometers' });
  return L.polygon(
    [left, tip, right].map(p => [p.geometry.coordinates[1], p.geometry.coordinates[0]] as [number,number]),
    { color:'#8E44AD', fillColor:'#8E44AD', fillOpacity:0.18, weight:1.5, dashArray:'6,4' }
  ).addTo(map);
}




5.5 directions.ts — Walking Directions List [NEW]
Renders the 7-step walking directions in the sidebar. Steps are semi-dynamic: step 3 and step 5 interpolate the shelter name and depth level.


export function renderWalkingDirections(
  shelter: ShelterFeature,
  container: HTMLElement
): void {
  const steps = [
    'Head west on the nearest street',
    'Continue straight — look for shelter signs',
    `Arrive at ${shelter.properties.name}`,
    'Enter immediately and go as deep underground as possible',
    `Move to the lowest level — ${shelter.properties.depthLevels} levels underground`,
    'Stay away from windows and outer walls',   // rendered in red
    'Wait for official all-clear before leaving',
  ];
  container.innerHTML = steps
    .map((s, i) => `<li class='${i===5 ? "danger" : ""}'>${s}</li>`)
    .join('');
}




6. Map Configuration [UPDATED]
6.1 Tile Layer
The live build uses OpenStreetMap standard tiles — not Mapbox dark tiles. This avoids needing a Mapbox token and is fully free with no registration.


export function initMap(containerId: string): L.Map {
  const map = L.map(containerId, {
    center: [40.7128, -74.0060],  // NYC
    zoom: 11,
    zoomControl: true,
  });
  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    maxZoom: 19,
    attribution: '© OpenStreetMap contributors'
  }).addTo(map);
  return map;
}


Zoom level 11 (vs 13 in v1 spec) gives a wider city view — important so users can see blast zones for 100kt and 1mt yields which extend far beyond Manhattan.

	Approach
	Time to analysis render
	< 2 seconds
